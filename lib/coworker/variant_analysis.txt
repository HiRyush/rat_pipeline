### parameters
--min-coverage 10 # 최소 read depth
--min-qual 30 # 최소 QUAL score
--min-alt-ratio 0.3 # target에서 최소 alt allele 비율
--max-ref-alt-ratio 0.1 # control 에서 최대 혀용 alt ratio
--min-recurrence 2 # 최소 반복 출현 샘플 수 
--min-delta-alt-ratio 0.2 # target-control alt ratio 최소 차이
